module:

module = MODULE ident ";" [ImportList] DeclarationSequence [BEGIN StatementSequence] END ident "." .

Some examples:

MODULE FOR_EXAMPLE;
    IMPORT Out;
    VAR i: INTEGER; foo: REAL;
BEGIN
    FOR i := 0 TO 10 DO
        Out.Int(i, 2);
    END
END FOR_EXAMPLE.
MODULE FOR_EXAMPLE;
    VAR i: INTEGER;
BEGIN
    foo;
END FOR_EXAMPLE.
